export const GET_ROOMS_QUERY = gql`
  query rooms(
    $capacity: Int
    $location: String!
    $office: String!
    $page: Int!
    $perPage: Int!
  ) {
    allRooms(
      page: $page
      perPage: $perPage
      capacity: $capacity
      location: $location
      office: $office
    ) {
      rooms {
        id
        name
        capacity,
        roomLabels,
        roomType,
        imageUrl,
      }
      pages
      queryTotal
      hasNext
      hasPrevious
    }
  }`;
